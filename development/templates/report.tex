\newcommand{\progressBar}[3]{%
  % #1=width, #2=fraction (0..1), #3=color
  \begin{tikzpicture}[baseline=-2pt]
    \fill[gray!15, rounded corners=2pt] (0,0) rectangle (#1,0.28);
    \fill[#3, rounded corners=2pt] (0,0) rectangle ({#1*#2},0.28);
    \node[font=\tiny\bfseries, white] at ({#1*#2/2},0.14) {\pgfmathparse{#2*100}\pgfmathprintnumber[fixed, precision=0]{\pgfmathresult}\%};
  \end{tikzpicture}%
}
